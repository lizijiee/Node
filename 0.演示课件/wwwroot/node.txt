npm  

//安装模块

npm install 模块名  

--save  //安装到生产环境（默认就是生产环境）

--save-dev  // 安装到开发环境（只能在本地运行）

-g  //安装到全局中

//卸载模块

npm  uninstall  模块名

npm init  //初始化一个项目


nrm 

//安装 nrm

npm install nrm -g  //安装到全局目录

nrm ls  //查看所有服务器

nrm use taobao  //通过use 切换服务器

nrm test  //测速
